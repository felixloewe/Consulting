\begin{frame}
	\frametitle{Einführung in die Graphentheorie}
	\textbf{Notation:}
	\begin{itemize}
		\item \textbf{$G = (V,E)$} ... ein \emph{Graph}
		\pause
		\item \textbf{$V = \left\{1,...,N_V\right\}$} ... Menge der \emph{Knoten}
		\pause
		\item \textbf{$E = \left\{(i,j)|i,j \in V, i\neq j\right\}$} ... Menge der \emph{Kanten}
		\pause
		\item $A \in N_V \times N_V$ ... eine \emph{Nachbarschaftsmatrix} \\ \vspace{0.2cm}
	$	a_{ij} =
		\begin{cases}
		  1 \text{ , } ij \in E \\
      0 \text{ , } ij  \notin E \\
   \end{cases}
	$
	\end{itemize}
	\vfill
	\pause
	\textbf{Begriffe:}
	\begin{itemize}
		\item \emph{Gerichteter} vs. \emph{ungerichteter} Graph
		\pause
		\item (In-/Out-) \emph{Degree}
		\pause
		\item \emph{Dichte}: $den(G) = \frac{|E_G|}{N_V(N_V-1)/2}$
	\end{itemize}

\end{frame}

\subsection{ERGM - Exponential Random Graph Model}
\begin{frame}
	\frametitle{ERGM - Exponential Random Graph Model}
	\begin{equation}
	\LARGE
	P_{\theta, \mathcal{X}}(X = x) = \frac{exp\left\{\theta^T g(x)\right\}}{\kappa(\theta, \mathcal{X})}
	\label{eq:ergm}
	\end{equation}
	mit
	\begin{itemize}
		\item $X$ zufällige Nachbarschaftsmatrix
		\item $x \in \mathcal{X}$, Menge aller möglichen Netzwerke
		\item $\theta \in \Omega \subset \mathbb{R}^q$ ... Vektor der Modellparameter
		\item $g(x)$ ... q-Vektor aus Statistiken basierend auf der Nachbarschaftsmatrix $x$ \\\vspace{1cm}
		\item \textcolor[rgb]{1,0,0}{\textbf{Problem:}}  $\kappa(\theta, \mathcal{X}) = \sum_{x \in \mathcal{X}}{exp			 \left\{\theta^T g(x)\right\}}$
	\end{itemize}
\end{frame}

\subsubsection{Simulation von Zufallsgraphen}
\begin{frame}
	\frametitle{Simulation von Zufallsgraphen}
	\textbf{Simulation einer Sequenz von Graphen aus Zielverteilung $P_\theta(x)$ via Makrov Chain Monte Carlo Algorithmus:}
	\vfill
	\begin{enumerate}
	\pause
	\item Beliebiges Netzwerk mit fester Knotenzahl N als Startpunkt. 
	\pause
	\item Aus dem aktuellen Graphen $x^{(m-1)}$ wird ein zufälliges Knotenpaar $i,j$ ($i,j \in 1, ...,N$) ausgewählt.
	\pause
	\item Vorgeschlagener Graph: $x^* = x^{(m-1)}$ bis auf $x_{ij}^{(m-1)} = 1 - x_{ij}^{(m-1)}$.
	\pause
	\item Akzeptanz mit der Wahrscheinlichkeit $min\{1, \frac{P_{\theta}(x^*)}{P_{\theta}(x^{(m-1)})}\}$.
	\pause
	\item Bei Akzeptanz $x^{(m)} = x^*$ und $x^{(m)} = x^{(m-1)}$ sonst.
	\item Iteration der Schritte 2 - 5.
\end{enumerate}
\pause
\begin{itemize}
	\item Algorithmus ist unabhängig vom Startpunkt bei ausreichendem \emph{Burn In}.
	\item Algorithmus ermöglicht unabhängige Ziehungen aus gleicher Kette durch \emph{Thinning}.
\end{itemize}
\end{frame}

\subsubsection{Schätzung der Modellparameter}
\begin{frame}
	\frametitle{Schätzung der Modellparameter}
Ziel: Zentrierung der Statistiken der simulierten Netzwerke über denen des beobachteten Netzwerkes:

\begin{equation}
\LARGE
	E_\theta(g(X)) - g(x_{obs}) = 0
	\label{eq:1}
\end{equation}

\begin{itemize}
	\item \textbf{\textcolor[rgb]{1,0,0}{Problem:}} $E_\theta(g(X)) = \sum_{x \in \mathcal{X}}{g(x)P_\theta(x)}$
	\vfill
	\item \textbf{\textcolor[rgb]{0.2,0.8,0.2}{Lösung}}: \emph{Importance Sampling}
	\pause
	\begin{enumerate}
		\item Ziehung einer großen Stichprobe von Graphen auf Basis eines vorläufigen Parametervektors $\tilde{\theta}$.
		\pause
		\item Benutzung gewichteter Stichprobendurchschnitte der Statistiken.
		\pause
		\item Erzeugen einer Sequenz von Parametern $\widetilde{\theta},\theta^{(1)}, \theta^{(2)},...,\theta^{(G)}$ durch \emph{Fisher Scoring}.
		\pause
		\item Neustart mit $\theta^{(G)}$ als $\tilde{\theta}$.
	\end{enumerate}

	
\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Degeneration}
	\textcolor[rgb]{1,0,0}{\textbf{Problem :}} Hohe Wahrscheinlichkeit auf unrealistischen Netzwerken (z.B. volles oder leeres Netzwerk) führt zur Divergenz des Schätz-Algorithmus. \\ \vfill
	\pause
	\textbf{Ursachen:}
	\begin{itemize}
		\item Instabilität von einfachen Zählstatistiken
		\item fehlende exogene Unterscheidungsmerkmale für Knoten und Kanten
		\item Beschränkung auf lineare Effekte der Statistiken unrealistisch
	\end{itemize}
	\vfill
	\pause
	\textcolor[rgb]{0,0.58,0}{\textbf{Lösungsansatz:}}
	\begin{itemize}
		\item Aufnahme von exogenen Kovariablen
		\item Aufnahme von nicht linearen Einflüssen durch\\ \emph{Curved Exponential Family Models}  
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Curved Exponential Family Models}
	\begin{itemize}
		\item \textbf{\textit{Geometrically Weighted Degree} (GWD):}
			\begin{equation}
				u(x, \phi_s) = e^{\phi_s} \sum^{n-1}_{i=1}\left\{1-(1-e^{-\phi_s})^{i}\right\}D_i(x)
			\label{eq:GWD}
			\end{equation}
						
			\begin{itemize}
				\item Kombination aus Zählstatistiken $D_i(x)$
				\item Abhängig von zusätzlichen Decay-Parameter $\phi$
			\end{itemize}
			\pause
			
		\item \textbf{\textit{Geometrically Weighted Edgewise Shared Partners} (GWESP):}
			\begin{equation}
				v(x, \phi_t) = e^{\phi_t} \sum^{n-2}_{i=1}\left\{1-(1-e^{-\phi_t})^{i}\right\}EP_i(x)
				\label{eq:GWESP}
			\end{equation}
		\item \textbf{\textit{Geometrically Weighted Dyadic Shared Partners} (GWDSP):}
			\begin{equation}
				w(x, \phi_p) = e^{\phi_p} \sum^{n-2}_{i=1}\left\{1-(1-e^{-\phi_p})^{i}\right\}DP_i(x)
			\label{eq:GWDSP}
			\end{equation}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Interpretation der Parameter}
	\textbf{Annahme:}\\
	$X^{(1)}$ und $X^{(2)}$ seien Netzwerke mit identischen Statistiken bis auf Statistik $g_i(X)$ und
	$$\delta_i(X) = g_i(X^{(1)}) - g_i(X^{(2)})$$. \vfill
	\textbf{Dann gilt:}
	$$ \frac{P(X^{(1)})}{P(X^{(2)})} = exp(\theta_i \delta_i(X)) $$ \vfill
	\pause
	$\Longrightarrow$ für positives $\delta_i(X)$ gilt also:
	\begin{itemize}
		\item Ist $\theta_i > 0$, so ist $X^{(1)}$ plausibler als $X^{(2)}$.
		\item Ist $\theta_i = 0$, so sind sie gleich plausibel.
		\item Ist $\theta_i < 0$, so ist $X^{(1)}$ plausibler als $X^{(2)}$.
	\end{itemize}
\end{frame}

	\frametitle{Interpretation der Parameter 3}
	\textbf{Curved ERGM Terms am Beispiel von Geometrically Weighted Degree (GWD):}
	\pause
	\begin{itemize}
		\item Änderung der Degree Sequenz eines Knotens: \\
		$(D_k, D_{k+1}) \rightarrow (D_k-1, D_{k+1}+1)$
		\pause
		\item $\frac{p_{after}}{p_{before}} = exp(\theta\rho^k)$ , $\rho = 1 - e^{-\phi}$
	\end{itemize}
	\pause
	\vfill
	$\Rightarrow$ \textbf{Interpretation der beiden Parameter:}
	\begin{itemize}
	\pause
		\item $\theta > 0$: Tendenz zum Hinzufügen von Kanten
		\item $\theta < 0$: Tendenz zum Löschen von Kanten Kanten
		\pause
		\item $\phi = 0$: Tendenz verschwindet komplett
		\item $\phi \rightarrow \infty$: Tendenz bleibt konstant
	\end{itemize}
\end{frame}

\frametitle{Probleme bei der Modellierung}
	\begin{itemize}
		\item Instabilität bezüglich Wahl der Statistiken, Jahr und Decay-Parameter
		\item Sehr lange Rechenzeit bei Einbindung von Kantenattributen und freiem Decay-Parameter
		\item Wahl der richtigen Statistiken schwierig / nicht eindeutig
	\end{itemize}
	\vfill
	\textbf{Idee:} Anwendung von Modell aus Vorgängerarbeit (Großwaffenhandel) auf Daten des Kleinwaffenhandels\\
	$\Longrightarrow$ funktioniert in keinem der 20 Jahre.\\\vfill
	$\Longrightarrow$ Netzwerke des Kleinwaffenhandels und Großwaffenhandels haben strukturelle Unterschiede
\end{frame}

	\textbf{Fazit}:	
	\begin{itemize}
		\item Zentrale Akteure dominieren den Handel
		\item Trend: mehr Handel, mehr beteiligte Nationen, größere Ausgaben
		\item Modellierung mit ERGM schwierig
	\end{itemize}

	\begin{itemize}
		\item Einbeziehung zusätzlicher Kovariablen

		\item Berücksichtigung der zeitlichen Struktur durch Temporal Exponential Random Graph Model (TERGM)
		\item Modellierung von Teilnetzwerken
	\end{itemize}
